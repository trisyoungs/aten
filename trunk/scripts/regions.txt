#
# Disordered Regions Example
#

# Use Universal FF for VDW parameters
loadff data/ff/uff.ff uff

# Create component models (all single atoms of different types)
newmodel blue
addatom N
ffmodel

newmodel red
addatom O
ffmodel

newmodel yellow
addatom Au
ffmodel

# Create box to MC in to
newmodel regions
setcell 20.0 20.0 20.0 90. 90. 90.
ffmodel

# Request components and set their regions
addcomponent cube blue 100
setoverlap blue true
setcentre blue 10 10 10
setshape blue cuboid
setgeometry blue 20.0 10.0 10.0

addcomponent bulk red 250
setoverlap red false
printcomponents

vdwscale 1.0

# Begin
disorder 1
