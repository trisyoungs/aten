#
# Disordered Regions Example
#

# Use Universal FF for VDW parameters
loadff("uff.ff", "uff");

# Create component models (all single atoms of different types)

# 'blue' will be contained in an orthorhomboid spanning the cell in the x-direction
newmodel("blue");
newatom(N);
nmols(1000);
setregion("cuboid", 15, 15, 15, 30, 15, 15, TRUE);

# 'red' will be allowed anywhere except 'blue's orthorhomboid
newmodel("red");
newatom(O);
nmols(2500);
regionoverlap(FALSE);

# 'yellow' will be allowed anywhere in the whole box (including 'blue's region)
newmodel("yellow");
newatom(Au);
nmols(2500);

# Create box to MC in to
newmodel("regions");
cell(30.0, 30.0, 30.0, 90, 90, 90);

# Begin
disorder(100);
