#
# [dmim]Cl 128 ion pair energy test
#
energyunits kj

# Load forcefield and set zmapping to use forcefield atom names
loadff data/ff/cldp-il.ff ilff

# Load model and associate forcefield
loadmodel dlpolytest/dmimcl128/dmimcl128.CONFIG il
ffmodel
chargeff
printmodel

# Set up energy expression
createexpression
elec ewald 0.27647 16 16 16
vcut 13.0
ecut 13.0

# Open trajectory
loadtrajectory dlpolytest/dmimcl128/dmimcl128.HISu

# Calculate model energy and print totals
modelenergy
printenergy
printewald
printelec
printvdw

quit
